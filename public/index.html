import requests

API_KEY = "brk_your_key_here"
BASE    = "https://bearcaptcha.live/api"

# one call — returns verified token
resp  = requests.post(
    f"{BASE}/captcha/solve",
    headers={"X-API-Key": API_KEY},
    json={
        "pageurl":   "https://target-site.com/login",
        "publickey": "YOUR-ARKOSE-PUBLIC-KEY"
    },
    timeout=60
)
token = resp.json()["token"]

# check remaining balance
bal = requests.get(
    f"{BASE}/captcha/credits",
    headers={"X-API-Key": API_KEY}
).json()["credits"]
print(f"Token: {token[:30]}...  |  Credits left: {bal}")
curl -X POST https://bearcaptcha.live/api/captcha/solve \
  -H "X-API-Key: brk_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"pageurl":"https://target-site.com/login","publickey":"YOUR-ARKOSE-PUBLIC-KEY"}'

# response
{
  "success": true,
  "token": "session_token|...",
  "credits_remaining": 499
}
const res = await fetch("https://bearcaptcha.live/api/captcha/solve", {
  method: "POST",
  headers: {
    "X-API-Key": "brk_your_key_here",
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    pageurl: "https://target-site.com/login",
    publickey: "YOUR-ARKOSE-PUBLIC-KEY"
  })
});
const { token } = await res.json();
console.log(token.slice(0, 30) + "...");
import asyncio, websockets, json

async def solve():
    uri = "wss://bearcaptcha.live/ws"
    async with websockets.connect(uri) as ws:
        await ws.send(json.dumps({"event": "auth", "api_key": "brk_key"}))
        await ws.recv()  # auth_ok

        await ws.send(json.dumps({
            "event": "solve",
            "request_id": "1",
            "pageurl": "https://target-site.com/login",
            "publickey": "YOUR-ARKOSE-PUBLIC-KEY"
        }))
        result = json.loads(await ws.recv())  # solving
        result = json.loads(await ws.recv())  # solved
        print(result["token"])

asyncio.run(solve())
